import mongoose from "mongoose";
import {DB_NAME} from "./constants";
import express from "express"
const app = express();

//iffy's
;(async()=>{

        await mongoose.connect(`${process.env.MONGODB_URI}/${DB_NAME}`)

        throw error
    }
})()
